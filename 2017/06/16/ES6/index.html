ES6 语法

ECMA是语法标准

let:

let定义变量 代码块 也就是{} 加个块级作用域；
1.具备块级作用域 其实就是立即执行函数调用
2.不能覆盖声明
3.解决i问题

const—用来定义常量

1.一旦赋值不能修改；
2.const必须赋初始值，因为以后没法赋值了；也不能重复声明
3.为了防止意外修改变量

var [a,b,c] = [12,5,101];
var {a,b,c} = {a:5,b:12,c:101}; //跟顺序无关

var [a,[b,c],d] = [2,[5,7],24]; //作用结构一样

function show(…args){
args.push(5);//原本args不是一个数组,加…就变为一个数组可以push
console.log(args);
}
show(1,2,3,4);

var map=new Map();
map.set(name,value); //设置
map.get(name);//获取
map.delete(name);//删除
//遍历map 不能用for in
for(var name of map){
console.log(name); // a,apple b,banana
}
for(var [key,value] of map){
console.log(key, value); // key value
}
for(var name of map.entries()){
console.log(name);
}
for(var key of map.keys()){ //只是循环key
console.log(key);
}
for(var val of map.values()){ //只是循环value
console.log(val);
}
